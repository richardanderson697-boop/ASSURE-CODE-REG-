          </div>

          <div className="mb-6">
            <h2 className="text-lg font-semibold mb-3">Compliance Deadlines</h2>
            <div className="space-y-3">
              {regulation.deadlines.map((deadline, i) => (
                <div key={i} className="flex items-start gap-3 p-4 border-l-4 border-orange-500 bg-orange-50 rounded">
                  <Calendar className="w-5 h-5 text-orange-600 flex-shrink-0 mt-0.5" />
                  <div>
                    <div className="font-semibold text-orange-900">{deadline.date}</div>
                    <div className="text-gray-700 mt-1">{deadline.description}</div>
                  </div>
                </div>
              ))}
            </div>
          </div>

          <div className="mb-6">
            <h2 className="text-lg font-semibold mb-3">Affected Entities</h2>
            <div className="grid grid-cols-2 gap-2">
              {regulation.affectedEntities.map((entity, i) => (
                <div key={i} className="flex items-center gap-2 p-2 bg-gray-50 rounded">
                  <Users className="w-4 h-4 text-gray-600" />
                  <span className="text-gray-700">{entity}</span>
                </div>
              ))}
            </div>
          </div>

          <div className="mb-6">
            <h2 className="text-lg font-semibold mb-3">Penalties for Non-Compliance</h2>
            <div className="p-4 bg-red-50 border-l-4 border-red-500 rounded">
              <AlertCircle className="w-5 h-5 text-red-600 mb-2" />
              <p className="text-gray-700">{regulation.penalties}</p>
            </div>
          </div>

          <div className="p-4 bg-gray-50 rounded-lg">
            <h2 className="text-sm font-semibold mb-2 text-gray-600">Source</h2>
            <a
              href={regulation.sourceUrl}
              target="_blank"
              rel="noopener noreferrer"
              className="text-blue-600 hover:text-blue-700 break-all"
            >
              {regulation.sourceUrl}
            </a>
          </div>
        </div>
      </div>
    );
  };

  return (
    <div className="min-h-screen bg-gray-50">
      {/* Header */}
      <div className="bg-white border-b border-gray-200 shadow-sm">
        <div className="max-w-7xl mx-auto px-4 py-4">
          <div className="flex items-center justify-between">
            <div className="flex items-center gap-3">
              <div className="w-10 h-10 bg-gradient-to-br from-blue-600 to-blue-700 rounded-lg flex items-center justify-center">
                <FileText className="w-6 h-6 text-white" />
              </div>
              <div>
                <h1 className="text-xl font-bold text-gray-900">Compliance Dashboard</h1>
                <p className="text-sm text-gray-600">Regulatory Intelligence & Monitoring</p>
              </div>
            </div>
            
            <div className="flex items-center gap-3">
              <div className="text-right mr-4">
                <div className="text-sm text-gray-600">Last Updated</div>
                <div className="text-sm font-medium">{new Date().toLocaleDateString()}</div>
              </div>
            </div>
          </div>
        </div>
      </div>

      {/* Main Content */}
      <div className="max-w-7xl mx-auto px-4 py-6">
        {selectedRegulation ? (
          <DetailView regulation={selectedRegulation} />
        ) : (
          <DashboardView />
        )}
      </div>

      {/* Footer */}
      <div className="max-w-7xl mx-auto px-4 py-4 mt-8">
        <div className="bg-blue-50 border border-blue-200 rounded-lg p-4 flex items-start gap-3">
          <AlertCircle className="w-5 h-5 text-blue-600 flex-shrink-0 mt-0.5" />
          <div className="text-sm text-blue-900">
            <strong>AI-Powered Analysis:</strong> All regulations are analyzed using advanced RAG (Retrieval-Augmented Generation) 
            to extract requirements, deadlines, and compliance frameworks. Download PDFs for detailed reports or integrate with 
            ASSURE-CODE for automated specification updates.
          </div>
        </div>
      </div>
    </div>
  );
};
